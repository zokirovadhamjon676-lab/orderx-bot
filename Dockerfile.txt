# Asosiy image – Python 3.10 ning yengil versiyasi
FROM python:3.10-slim

# Konteyner ichida ishchi papka yaratish
WORKDIR /app

# Tizim paketlarini o‘rnatish (kerak bo‘lsa)
RUN apt-get update && apt-get install -y --no-install-recommends \
    gcc \
    && rm -rf /var/lib/apt/lists/*

# Python kutubxonalarini o‘rnatish (avval requirements faylini nusxalaymiz)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Loyiha kodini konteynerga nusxalash
COPY . .

# Botni ishga tushirish
CMD ["python", "-m", "bot.main"]
